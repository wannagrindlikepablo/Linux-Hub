local PlaceId = game.PlaceId

if PlaceId ~= 11701792069 then
    -- Notify the user that the script can only run in the specified place
    Fluent:Notify({
        Title = "Error",
        Content = "This script can only be used in Snow Plow Simulator.",
        Duration = 5
    })
    return -- Stop the script if the place ID does not match
end

local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

local Window = Fluent:CreateWindow({
    Title = "Linux Hub Snow Plow Simulator", -- Updated window title
    SubTitle = "Toggle and Button Script",
    TabWidth = 200, -- Increased tab width for better visibility
    Size = UDim2.fromOffset(800, 600), -- Increased size for a larger UI
    Theme = "Dark"
})

local Tabs = {
    Main = Window:AddTab({ Title = "Main" }),
    Credits = Window:AddTab({ Title = "Credits" }) -- New Credits tab
}

local Options = Fluent.Options

-- Updated paragraph with instructions in the Main tab
Tabs.Main:AddParagraph({
    Title = "Instructions",
    Content = "You must press the 'Spawn the Vehicle' button to be able to use 'Op Snow Farm'."
})

-- Toggle to activate/deactivate continuous script for Bulldozer
local BulldozerToggle = Tabs.Main:AddToggle("ScriptToggle_Bulldozer", { Title = "Op Snow Farm for Bulldozer", Default = false })

BulldozerToggle:OnChanged(function()
    if Options.ScriptToggle_Bulldozer.Value then
        print("Op Snow Farm for Bulldozer activated")
        task.spawn(function()
            while Options.ScriptToggle_Bulldozer.Value do
                local ohInstance1 = workspace.Bulldozer
                local ohInstance2 = workspace.HitParts.Snow2
                local ohString3 = "Snow2"
                
                game:GetService("ReplicatedStorage").Events.uLg98Wsc4r5QT1hb:FireServer(ohInstance1, ohInstance2, ohString3)
                wait(0.1)
            end
        end)
    else
        print("Op Snow Farm for Bulldozer deactivated")
    end
end)

-- Toggle to activate/deactivate continuous script for SantasSleigh
local SantasSleighToggle = Tabs.Main:AddToggle("ScriptToggle_SantasSleigh", { Title = "Op Snow Farm for SantasSleigh", Default = false })

SantasSleighToggle:OnChanged(function()
    if Options.ScriptToggle_SantasSleigh.Value then
        print("Op Snow Farm for SantasSleigh activated")
        task.spawn(function()
            while Options.ScriptToggle_SantasSleigh.Value do
                local ohInstance1 = workspace.SantasSleigh
                local ohInstance2 = game:GetService("ReplicatedStorage").GroundSnow.HitParts3.HitParts.Snow3
                local ohString3 = "Snow3"

                game:GetService("ReplicatedStorage").Events.uLg98Wsc4r5QT1hb:FireServer(ohInstance1, ohInstance2, ohString3)
                wait(0.1) -- Small delay to prevent server overload
            end
        end)
    else
        print("Op Snow Farm for SantasSleigh deactivated")
    end
end)

-- Button to execute the Spawn Bulldozer script
Tabs.Main:AddButton({
    Title = "Spawn Bulldozer Robux Vehicle",
    Description = "Press to spawn the Bulldozer",
    Callback = function()
        local ohString1 = "Bulldozer"
        local ohString2 = "yes"
        
        game:GetService("ReplicatedStorage").Shop.Events.SpawnCar:FireServer(ohString1, ohString2)
        print("SpawnCar script executed")
    end
})

-- Button to execute the Spawn SantasSleigh script
Tabs.Main:AddButton({
    Title = "Spawn SantasSleigh Event Vehicle",
    Description = "Press to spawn SantasSleigh",
    Callback = function()
        local ohString1 = "SantasSleigh"
        local ohString2 = "yes"

        game:GetService("ReplicatedStorage").Shop.Events.SpawnCar:FireServer(ohString1, ohString2)
        print("Spawn SantasSleigh script executed")
    end
})

-- Button to execute the Inf Rebirth + Inf Snow script
Tabs.Main:AddButton({
    Title = "Inf Rebirth + Inf Snow",
    Description = "Press to execute Infinite Rebirth and Snow.\nTo get Inf Snow, just collect any snow after getting Inf Rebirth.\nNote: You cannot revert the rebirths back to normal.", -- Description with note
    Callback = function()
        -- Alert before executing the Inf Rebirth script
        Fluent:Notify({
            Title = "Warning",
            Content = "You cannot revert the rebirths back to normal!",
            Duration = 5
        })

        local args = {
            [1] = -2
        }
        
        game:GetService("ReplicatedStorage"):WaitForChild("RebirthEvent"):FireServer(unpack(args))

        wait(1)
        
        local args = {
            [1] = 1/0
        }

        game:GetService("ReplicatedStorage"):WaitForChild("RebirthEvent"):FireServer(unpack(args))
        print("Inf Rebirth + Inf Snow script executed")
    end
})

-- Add Credits section with Discord note
Tabs.Credits:AddParagraph({
    Title = "Credits",
    Content = "Developed by: _lxzr_\nDiscord Username"
})

Window:SelectTab(1)

Fluent:Notify({
    Title = "Linux Hub",
    Content = "Use the toggle for continuous activation or the button for a one-time action.",
    Duration = 5
})
