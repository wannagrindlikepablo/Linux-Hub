-- Add a dropdown to choose the aura
local AuraDropdown = MainTab:AddDropdown("AuraDropdown", {
    Title = "Choose Aura",
    Values = auraNames,
    Default = 1,
})

-- Add a button called "Get Aura"
MainTab:AddButton({
    Title = "Get Aura",
    Callback = function()
        -- Unequip all auras
        game:GetService("ReplicatedStorage").Remotes.AuraUnequipAll:FireServer()

        -- Get the selected aura from the dropdown
        local selectedAura = AuraDropdown.Value
        local ohInstance1 = game:GetService("ReplicatedStorage").Auras[selectedAura]

        -- Equip the selected aura
        game:GetService("ReplicatedStorage").Remotes.AuraEquip:FireServer(ohInstance1)
    end
})

-- Add a "Credits" tab
